-- Dynamic view 4, the create view will select all clients who live in the city 'ott'
CREATE VIEW Canada_V as
select * from clients
where city = 'ott';

-- Dynamic view 9, the create view materialized selects all clients with the name 'andrianna'

where name = 'andrianna';
