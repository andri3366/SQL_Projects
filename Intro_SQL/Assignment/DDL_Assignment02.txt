-- will drop the database if assignment02 exists and create a new database to use
drop database if exists assignment02;
create database assignment02;
use assignment02;

create table services ( -- creates table for service with primary key id
	id int not null auto_increment,
    name varchar(50) not null,
    description mediumtext,
    cost numeric(10,2),
    constraint servicesPK primary key (id)
    );
    
create table countries( -- creates table for couuntries with primary key id
	id int not null auto_increment,
    name varchar(50) not null,
    constraint countriesPK primary key (id)
);

create table states_provinces( -- creates table for states_provinces with primary key id and foreign key with referencce to countries
	id int not null auto_increment,
    name varchar(50) not null,
    country_id int,
    constraint states_provincesPK primary key (id),
    constraint countriesFK foreign key (country_id)
		references countries (id)
        on update no action on delete no action
        -- when the table is updated there is no changes to the table countries
);
    
create table clients ( -- creates table for clients with primary key id with references to states_provinces and countries
	id int not null auto_increment,
    name varchar(50) not null,
    address varchar(50),
    address2 varchar(50),
    city varchar(50),
    postal_code varchar(12),
    phone_number varchar(12),
    email varchar(150),
    states_province_id int,
    country_id int,
    constraint clientsPK primary key (id),
    constraint states_provincesFK foreign key (states_province_id)
		references states_provinces (id)
        on update no action on delete no action,
	constraint countryFK foreign key (country_id)
		references countries (id)
        on update no action on delete no action
);

create table sales_orders( -- creates table for slaes_orders with primary key id and reference to clients
	id int not null auto_increment,
    order_date datetime not null,
    invoice_date datetime,
    comments longtext,
    client_id int,
    constraint sales_ordersPK primary key (id),
    constraint clientsFK foreign key (client_id)
		references clients (id)
        on update no action on delete no action
);

create table sales_order_services( -- creates table for sales_order_services with primary key id with reference to sales_order, clients, and services
	id int not null auto_increment,
    quantity numeric,
    discount numeric,
    service_schedule_date datetime,
    extra_information longtext,
    sales_order_id int,
    service_id int,
    client_id int,
    constraint sales_order_servicesPK primary key (id),
    constraint sales_ordersFK foreign key (sales_order_id)
		references sales_orders (id)
        on update no action on delete no action, 
	constraint servicesFK foreign key (service_id)
		references services (id)
        on update no action on delete no action,
	constraint clientFK foreign key (client_id)
		references clients (id)
        on update no action on delete no action
);

create table history_types( -- creates table for history_types with primary key id
	id int not null auto_increment,
    name varchar(50),
    constraint history_typesPK primary key (id)
);

create table sales_order_service_history( -- creates table for sales_orders_service_history with primary key id with reference to sales_order_services and history_types
	id int not null auto_increment,
    entry_date datetime,
    notes longtext,
    sales_order_service_id int,
    history_type_id int,
    constraint sales_order_service_historyPK primary key (id),
    constraint sales_order_servicesFK foreign key (sales_order_service_id)
		references sales_order_services (id)
        on update no action on delete no action,
    constraint history_typesFK foreign key (history_type_id)
		references history_types (id)
        on delete no action
);
